186  void TCPClient<HEADER>::handleReadBody(const boost::system::error_code& error)
187  {
188  size_t queue_size = 0;
189 
190  if (kill_)
191  {
192  return;
193  }
194  else if (error)
195  {
196  std::cerr << "Error reading body: "
197  << error.message() << std::endl;
198  throw SocketReadError(error.message());
199  }
200  else
201  {
202  std::unique_lock<std::mutex> lk(buff_queue_mutex_);
203 
204  // copy into shared_ptr
205  buff_queue_.push(std::make_shared<std::vector<char>>(buff_));
206 
207  queue_size = buff_queue_.size();
208 
209  if (queue_size > max_queue_size_)
210  {
211  buff_queue_.pop();
212  lk.unlock();
213  std::cout << "Warning: Client dropped packet due to full buffer" << std::endl;
214  }
215  else
216  {
217  lk.unlock();
218  }
219 
220  // Free up the CPU to allow the consumer thread a chance to keep up.
221  if (queue_size > 1)
222  {
223  std::this_thread::yield();
224  }
225  else
226  {
227  // Consumer thread only waits to be notified when the queue is (was) empty
228  buff_queue_conditional_.notify_one();
229  }
230  }
231 
232  // get ready to read again
233  startDataRead();
234  }

236  template <class HEADER>
237  void TCPClient<HEADER>::signalPackets()
238  {
239  // define condition to continue: something in buffer or kill flag set
240  auto continue_condition = [this]{return (!buff_queue_.empty() || kill_);};
241 
242  for (;;)
243  {
244  std::unique_lock<std::mutex> lk(buff_queue_mutex_);
245  // if queue is empty, wait for something to be in the queue
246  if (!continue_condition())
247  buff_queue_conditional_.wait(lk, continue_condition);
248 
249  if (kill_)
250  return;
251 
252  decltype(buff_queue_) local_q;
253  std::swap(buff_queue_, local_q);
254  lk.unlock();
255 
256  while (!local_q.empty())
257  {
258  auto packet = local_q.front();
259  local_q.pop();
260  signal_(packet);
261  }
262  }
263  }
